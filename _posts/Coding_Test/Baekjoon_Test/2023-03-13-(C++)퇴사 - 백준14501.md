## 코드

```c++
#include<iostream>
using namespace std;

int inArr[20][2];
int outArr[20];

int N;
int i, j, k;
int result;

void dfs(int, int);

int main() {
	// input
	cin >> N;
	for (i = 0; i <N; i++) {
		cin >> inArr[i][0] >> inArr[i][1];
	}

	// run
	dfs(0, 0);

	// output
	cout << result;

	return 0;
}

void dfs(int depth, int value) {
	if (depth > N) return;

	result = max(result, value);
	for (int i = depth; i < N; i++) {
		dfs(i + inArr[i][0], value + inArr[i][1]);
	}
}
```
